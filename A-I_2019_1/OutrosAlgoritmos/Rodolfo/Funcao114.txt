using namespace std;
int devolveMaior(int matriz[3][3])
{
    int maior = matriz[0][0];
    for (int l=0; l < 3; l++)
    {
        for (int c=0; c < 3; c++)
        {
            if (matriz[l][c] > maior)
            {
                maior = matriz[l][c];
            }
        }
    }
    return maior;
}
bool verifica (int matriz[3][3],int numero)
{
    int condition = false;
    for (int l=0; l < 3; l++)
    {
        for (int c=0; c < 3; c++)
        {
            if (numero == matriz[l][c])
            {
                condition = true;
                break;
            }
        }
    }
    return condition;
}
void retornaDiag (int m[3][3])
{
    int x=0, v[3];
    for (int l=0; l < 3; l++)
    {
        for (int c=0; c < 3; c++)
        {
            if (l == c)
            {
                v[x] = m[l][c];
                x++;
            }
        }
    }
    for (int i=0; i < 3; i++)
    {
        cout << v[i] << endl;
    }
}
